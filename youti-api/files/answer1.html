class Solution{
    char firstTwiceChar(string s){
         if(s.size()<=1)return '';
         unsigned int hashTable[256];//字符是8bit类型，最多有256种，开辟1k空间当做hash表
         for(auto i = 0; i < 256; i++)hashTable[i] = 0;
         for(auto i = 0; i < s.size(); i++){
            if(++hashTable[s[i]] == 2)
                return s[i];
        }
    return '';//没有出现两次的字符
    }
}
